INSERT INTO employee_job_table (employee_id, job_id, effective_date)
SELECT employee_id, 4, '2020-01-01'
FROM employee_profile_table
WHERE employee_no = 1002;



SELECT e.employee_name, e.employee_no, ej.effective_date, j.job_title
FROM employee_profile_table e
JOIN employee_job_table ej ON e.employee_id = ej.employee_id
JOIN job_profile_table j ON ej.job_id = j.job_id;



SELECT e.employee_name, e.employee_no, j.job_name
FROM employee_profile_table e
JOIN employee_job_table ej ON e.employee_id = ej.employee_id
JOIN job_profile_table j ON ej.job_id = j.job_id
WHERE ej.effective_date = (
    SELECT MAX(ej2.effective_date)
    FROM employee_job_table ej2
    WHERE ej2.employee_id = e.employee_id
);



DELETE ej
FROM employee_job_table ej
JOIN employee_profile_table e ON ej.employee_id = e.employee_id
WHERE e.employee_no = 1000
AND ej.effective_date = '2020-01-01';
